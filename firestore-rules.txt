rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow public read access to blog posts
    match /posts/{document} {
      allow read: if true;
      // Only authenticated users can create/update/delete posts
      allow create, update, delete: if request.auth != null && request.auth.token.email == 'abdulhakimbd010@gmail.com';
    }
    
    // Allow public write access for service applications (customers don't need to login)
    match /serviceApplications/{document} {
      allow create: if true; // Anyone can submit applications
      allow read, update, delete: if request.auth != null && request.auth.token.email == 'abdulhakimbd010@gmail.com'; // Only admin can read/manage
    }
    
    // Allow public write access for contact messages
    match /contactMessages/{document} {
      allow create: if true; // Anyone can send messages
      allow read, update, delete: if request.auth != null && request.auth.token.email == 'abdulhakimbd010@gmail.com'; // Only admin can read/manage
    }
    
    // Allow public write access for mailing list subscriptions
    match /mailingList/{document} {
      allow create: if true; // Anyone can subscribe
      allow read, update, delete: if request.auth != null && request.auth.token.email == 'abdulhakimbd010@gmail.com'; // Only admin can read/manage
    }
  }
}
